# generate data
def step(x):
    return 0 if x < 0 else 1

def tanh(t):
    def f(x, t=t):
        return 1 - 2.0 / (1+math.exp(t * x))
    return f

def sigmoid(t):
    def f(x, t=t):
        return 1.0 / (1+math.exp(-x / t))
    return f

def relu(x):
    return 0 if x < 0 else x

def plot(x, y, color, title, filename):
    # setup plot
    fig, ax = plt.subplots()
    ax.plot(x,y,color)
    ax.set(
        title=title,
        ylabel='f(x)',
        xlabel='x'
    )

    plt.setp(ax.get_yticklabels(), rotation_mode="anchor", ha="right", va="center")
    plt.setp(ax.get_xticklabels(), rotation=0, rotation_mode="anchor", ha="center", va="top")

    # save and show
    svg_path = os.path.join("{}.svg".format(filename))
    plt.savefig(svg_path, format="svg", bbox_inches='tight')
    plt.show()
    
x = np.arange(-5, 5, 0.01)
